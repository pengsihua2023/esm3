## 自监督领域适应
```
import os
import torch
from Bio import SeqIO
from esm.pretrained import ESM3_sm_open_v0
from esm.tokenization.sequence_tokenizer import EsmSequenceTokenizer  # 导入序列分词器
from esm.utils.constants.esm3 import SEQUENCE_MASK_TOKEN  # 导入序列掩码代号

# 创建保存模型的目录
model_save_dir = './model-new'
os.makedirs(model_save_dir, exist_ok=True)

# 指定设备
device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')

# 载入预训练的 ESM3 模型
model = ESM3_sm_open_v0(device=device)

# 冻结部分层，只有 sequence_head 可训练
for name, param in model.named_parameters():
    if 'sequence_head' in name:
        param.requires_grad = True
    else:
        param.requires_grad = False

# 将模型设置为训练模式
model.train()

# 读取 FASTA 文件
train_path = './data/HA_train.fa'
test_path = './data/HA_test.fa'
train_sequences = [str(record.seq) for record in SeqIO.parse(train_path, "fasta")]
test_sequences = [str(record.seq) for record in SeqIO.parse(test_path, "fasta")]

# 初始化分词器
tokenizer = EsmSequenceTokenizer()

# 随机选择一些位置进行掩码，同时生成用于训练的“标签”
def mask_sequence(sequence, mask_prob=0.15):
    tokens = tokenizer.tokenize(sequence).tokens  # 将序列分词
    labels = [-100] * len(tokens)  # 初始化标签为 -100，表示该位置不用于损失计算
    for i in range(len(tokens)):
        if torch.rand(1) < mask_prob:
            labels[i] = tokens[i]  # 掩码位置的原始字符索引作为标签
            tokens[i] = tokenizer.token_to_id(SEQUENCE_MASK_TOKEN)  # 使用掩码标记
    return tokens, torch.tensor(labels, dtype=torch.long)

# 将数据处理成模型可以接受的格式
def collate_batch(batch):
    batch_tokens, batch_labels = [], []
    for seq in batch:
        tokens, labels = mask_sequence(seq)
        batch_tokens.append(torch.tensor(tokens, dtype=torch.long))
        batch_labels.append(labels)
    batch_tokens = torch.nn.utils.rnn.pad_sequence(batch_tokens, batch_first=True, padding_value=tokenizer.pad_token_id)
    batch_labels = torch.nn.utils.rnn.pad_sequence(batch_labels, batch_first=True, padding_value=-100)
    return batch_tokens, batch_labels

# 简单的训练循环
optimizer = torch.optim.Adam(filter(lambda p: p.requires_grad, model.parameters()), lr=0.0005)
for epoch in range(5):  # 训练轮次
    for i in range(0, len(train_sequences), 8):  # 批次大小为 8
        batch_tokens, batch_labels = collate_batch(train_sequences[i:i+8])
        optimizer.zero_grad()
        outputs = model(batch_tokens, repr_layers=[33])  # 只获取最后一层的输出
        logits = outputs['logits']
        loss = torch.nn.functional.cross_entropy(logits.view(-1, logits.size(-1)), batch_labels.view(-1), ignore_index=-100)
        loss.backward()
        optimizer.step()
        print(f"Epoch {epoch+1}, Batch {i//8+1}, Loss: {loss.item()}")

    # 验证循环
    model.eval()
    with torch.no_grad():
        total_loss = 0
        count = 0
        for seq in test_sequences:
            test_tokens, test_labels = collate_batch([seq])
            outputs = model(test_tokens, repr_layers=[33])
            logits = outputs['logits']
            loss = torch.nn.functional.cross_entropy(logits.view(-1, logits.size(-1)), test_labels.view(-1), ignore_index=-100)
            total_loss += loss.item()
            count += 1
        average_loss = total_loss / count
        print(f"Epoch {epoch+1}, Validation Loss: {average_loss}")

    # 返回到训练模式
    model.train()

    # 保存模型
    model_save_path = os.path.join(model_save_dir, f'esm3_epoch_{epoch+1}.pt')
    torch.save(model.state_dict(), model_save_path)
    print(f"Model saved to {model_save_path}")

print("Training complete!")
```
